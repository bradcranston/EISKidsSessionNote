/* SIGNATURE HANDLING
   
   Signatures are stored as base64-encoded PNG images (data URLs).
   
   To save signatures from web viewer to FileMaker container fields:
   1. Extract the base64 data from the returned JSON
   2. Use Base64Decode() to convert to binary
   3. Insert into container field
   
   Example script steps:
   Set Variable [ $json ; Value: Get(ScriptParameter) ]
   Set Variable [ $eiSig ; Value: JSONGetElement($json ; "eISignatureBase64") ]
   Set Variable [ $parentSig ; Value: JSONGetElement($json ; "parentSignatureBase64") ]
   
   # Remove the "data:image/png;base64," prefix before decoding
   Set Variable [ $eiSigData ; Value: Substitute($eiSig ; "data:image/png;base64," ; "") ]
   Set Variable [ $parentSigData ; Value: Substitute($parentSig ; "data:image/png;base64," ; "") ]
   
   # Insert into container fields
   Set Field [ SessionNote::EISignature ; Base64Decode($eiSigData ; "signature.png") ]
   Set Field [ SessionNote::ParentSignature ; Base64Decode($parentSigData ; "signature.png") ]
   
   To load signatures from FileMaker to web viewer:
   Use Base64Encode() with the "data:image/png;base64," prefix
   
   Set Field [ SessionNote::EISignatureBase64 ; 
     "data:image/png;base64," & Base64Encode(SessionNote::EISignature) ]
*/
